205static inline int calculate_accurate_MBR(uint8_t sensor_num, int val)
206{ // for better accuracy, enlarge SDR to two byte scale
207 if (SDR_M(sensor_num) == 0) {
208 return ((val << 8) * SDR_Rexp(sensor_num));
209 }
210 return ((val << 8) / SDR_M(sensor_num) * SDR_Rexp(sensor_num));
211}
212
213static inline int calculate_MBR(uint8_t sensor_num, int val)
214{
215 if (SDR_M(sensor_num) == 0) {
216 return (val * SDR_Rexp(sensor_num) + round_add(sensor_num, val));
217 }
218 return (val * SDR_Rexp(sensor_num) / SDR_M(sensor_num) + round_add(sensor_num, val));
219}
220
221static inline float convert_MBR_to_reading(uint8_t sensor_num, uint8_t val)
222{
223 if (SDR_M(sensor_num) == 0) {
224 return (val - round_add(sensor_num, val)) / SDR_Rexp(sensor_num);
225 }
226
227 return (val - round_add(sensor_num, val)) * SDR_M(sensor_num) / SDR_Rexp(sensor_num);
228}
